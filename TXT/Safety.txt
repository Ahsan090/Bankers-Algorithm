def is_work_safe(work):
    for p in work:
        if p.is_finished == False:
            return False
    return True

def safety_walkthrough(Processes, available):
    """
    Returns (is_safe, sequence, steps).
    steps is a list of dicts — one per process-check (fits AND skips) across all passes.
    Each dict: pass_num, pid, allocation, pmax, need,
               available_before, fits (bool), available_after.
    Does NOT mutate process objects.
    """
    scheduled = set()
    sequence = []
    steps = []
    pass_num = 1
    while len(scheduled) < len(Processes):
        found_pid = None
        for p in Processes:
            if p.pid in scheduled:
                continue
            avail_before = available[:]
            fits = all(p.need[j] <= available[j] for j in range(len(available)))
            if fits:
                avail_after = [available[j] + p.allocation[j] for j in range(len(available))]
                steps.append({'pass_num': pass_num, 'pid': p.pid,
                               'allocation': p.allocation[:], 'pmax': p.pmax[:],
                               'need': p.need[:], 'available_before': avail_before,
                               'fits': True, 'available_after': avail_after})
                available = avail_after
                sequence.append(p.pid)
                found_pid = p.pid
                pass_num += 1
                break
            else:
                steps.append({'pass_num': pass_num, 'pid': p.pid,
                               'allocation': p.allocation[:], 'pmax': p.pmax[:],
                               'need': p.need[:], 'available_before': avail_before,
                               'fits': False, 'available_after': avail_before})
        if found_pid:
            scheduled.add(found_pid)
        else:
            break
    return len(scheduled) == len(Processes), sequence, steps

def safety(Processes, available):
    work = Processes.copy()
    for p in work:
        p.is_finished = False
    sequence = []
    while True:
        progress = False
        for p in work:
            if all(p.need[j] <= available[j] for j in range(len(available))):
                available = [available[j] + p.allocation[j] for j in range(len(available))]
                p.is_finished = True
                sequence.append(p.pid)
                work.remove(p)
                progress = True
                break
        if not progress:
            break

    return is_work_safe(work), sequence
